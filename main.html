function secondHighest(arr) { if (arr.length < 2) { return -Infinity; } // Convert input to numbers arr = arr.map(Number); // Sort in ascending order arr.sort((a, b) => a - b); // Find the second highest unique element let highest = arr[arr.length - 1]; for (let i = arr.length - 2; i >= 0; i--) { if (arr[i] < highest) { return arr[i]; } } return -Infinity; // If all elements are the same } function Main() { let n = parseInt(prompt("Enter the number of elements")); let arr = []; for (let i = 0; i < n; i++) { arr[i] = prompt("Enter element " + (i + 1)); } alert(secondHighest(arr)); } Main();
